            <p class="max-w-3xl mx-auto text-lg text-gray-600">

                This interactive guide breaks down Louisiana House Bill 549 (Act 19). The goal is to enhance public safety, reduce fraud, and lower claim costs for everyone.

                        <h4 class="font-bold text-xl mb-4">Your Responsibilities as a Policyholder</h4>

                        <ul class="space-y-3 text-gray-600 list-disc list-inside">

                           <li>Install and maintain compliant dash cam and telematics systems on all relevant fleet vehicles.</li>

                           <li>Provide proof of installation and operation to your insurer (e.g., vendor certificate).</li>

                           <li>Ensure the systems are in continuous use throughout the policy period.</li>

                           <li>Annually, submit a telematics summary report or a signed affidavit to your insurer to verify ongoing compliance.</li>

                           <li>Failure to maintain the systems will result in losing the discount at the next renewal.</li>

                        </ul>

                    </div>

                    <div id="insurer-content" class="tab-pane hidden">

                        <h4 class="font-bold text-xl mb-4">Insurer Responsibilities</h4>

                         <ul class="space-y-3 text-gray-600 list-disc list-inside">

                           <li>Provide an actuarially justified liability premium discount for all compliant commercial vehicles.</li>

                           <li>Annually verify that policyholders remain in compliance with the technology requirements.</li>

                           <li>By March 1st each year, report to the Commissioner of Insurance on the number of discounted policies, total savings, and observed changes in claim data.</li>

                           <li>Face penalties for willful noncompliance with the law.</li>

                        </ul>

                    </div>

                        <h4 class="font-bold text-xl mb-4">Commissioner of Insurance Responsibilities</h4>

                        <ul class="space-y-3 text-gray-600 list-disc list-inside">

                           <li>Establish and publish the minimum technical standards for dash cams and telematics systems.</li>

                           <li>Create rules for verification procedures and potential exemptions.</li>

                           <li>Compile annual reports from all insurers into a summary for the state legislature.</li>

                           <li>The first summary report to the legislature is due by June 1, 2027.</li>

                        </ul>

                    tabButtons.forEach(btn => btn.classList.remove('active'));

                    button.classList.add('active');


                    const tab = button.getAttribute('data-tab');

                    tabPanes.forEach(pane => {

                        pane.classList.add('hidden');

                        pane.classList.remove('active');

                    });

                    

                    const activePane = document.getElementById(`${tab}-content`);

                    activePane.classList.remove('hidden');

                    activePane.classList.add('active');

                });

            });

            

            const calculateBtn = document.getElementById('calculate-btn');

            const premiumInput = document.getElementById('premium-input');

            const chartExplainer = document.getElementById('chart-explainer');

            const ctx = document.getElementById('savingsChart').getContext('2d');

            let savingsChart;


            function createOrUpdateChart(premium = 0, savings = 0) {

                const remaining = premium - savings;

                const savingsPercentage = premium > 0 ? ((savings / premium) * 100).toFixed(1) : 0;

                

                chartExplainer.textContent = `With a ${savingsPercentage}% estimated discount, you could save $${savings.toLocaleString()}.`;


                if (savingsChart) {

                    savingsChart.data.datasets[0].data = [remaining, savings];

                    savingsChart.update();

                } else {

                    savingsChart = new Chart(ctx, {

                        type: 'doughnut',

                        data: {

                            labels: ['Remaining Premium', 'Potential Savings'],

                            datasets: [{

                                data: [remaining, savings],

                                backgroundColor: [

                                    '#d1d5db',

                                    '#2563eb'

                                ],

                                borderColor: '#F8F7F4',

                                borderWidth: 4

                            }]

                        },

                        options: {

                            responsive: true,

                            maintainAspectRatio: false,

                            cutout: '70%',

                            plugins: {

                                legend: {

                                    position: 'bottom',

                                },

                                tooltip: {

                                    callbacks: {

                                        label: function(context) {

                                            let label = context.label || '';

                                            if (label) {

                                                label += ': ';

                                            }

                                            if (context.parsed !== null) {

                                                label += new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(context.parsed);

                                            }

                                            return label;

                                        }

                                    }

                                }

                            }

                        }

                    });

                }

            }

            

            createOrUpdateChart();

            chartExplainer.textContent = 'Enter your premium to see a potential savings breakdown.';


            calculateBtn.addEventListener('click', () => {

                const premium = parseFloat(premiumInput.value);

                if (!isNaN(premium) && premium > 0) {

                    const estimatedDiscount = 0.10; 

                    const savings = premium * estimatedDiscount;

                    createOrUpdateChart(premium, savings);

                } else {

                   createOrUpdateChart();

                   chartExplainer.textContent = 'Please enter a valid premium amount.';

                }

            });

        });
